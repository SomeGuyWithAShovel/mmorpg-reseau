# FOR WINDOWS POWERSHELL
#
# curl.exe instead of curl
# ` is the escape character, used to have a single command in multiple lines
# "\`"" is an escaped " surrounded by " (it would be "\"" in C/C++)

curl.exe -i "http://localhost:3000"

curl.exe -i "http://localhost:3000/health"

curl.exe -i "http://localhost:3000/login" `
-H "Content-Type: application/json" `
-d "{\`"username\`":\`"testname\`",\`"password\`":\`"testpassword\`"}"

curl.exe -i "http://localhost:3000/login" `
-H "Content-Type: application/json" `
-d "{\`"username\`":\`"testname\`",\`"password\`":\`"1234\`"}"
